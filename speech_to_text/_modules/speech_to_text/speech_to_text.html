[docs]def audio_transcribe(audio_file, auth_file, save_dir, prefix,
                     language = 'en-US_BroadbandModel', watson_learn = False):
    """
    Takes a wav audio file and uses IBM's speech to text services to transcribe
    the audio into text and returns the individual words found in the audio along
    with names of the audio snippet files that were created for each word.

    Args:
        audio_file (str): Name of location of audio file to be transcribed.
        auth_file (str): Name of location of .json file containing credentials needed to
        use the speech to text service from IBM.
        save_dir (str): Name of location in which audio snippets for individual
        words should be saved.
        prefix (str): Prefix to be given to each individual audio snippet file.
        language (str): String indicating language for transcription. See Languages_
        for more information. Default is US English broadband.
        watson_learn (boolean): Boolean indicating whether or not information should
        be sent to IBM to improve their services. Default is False.

    .. _Languages: https://www.ibm.com/watson/developercloud/doc/speech-to-text/input.html#models

    Returns:
        dict: Dictionary with individual words and the names of audio files
        containing the audio snippets for each word.
    """

    import json
    from watson_developer_cloud import SpeechToTextV1

    with open(auth_file) as cred_file:
        auth_info = json.load(cred_file)

    speech_user = auth_info['username']
    speech_pass = auth_info['password']

    speech_to_text = SpeechToTextV1(
        username = speech_user,
        password = speech_pass,
        x_watson_learning_opt_out = watson_learn
    )

    speech_to_text.get_model(language)

    with open(audio_file, 'rb') as audio:
        speech_results = speech_to_text.recognize(
            audio, content_type = 'audio/wav', timestamps = True)

    alternatives = speech_results['results']
    num_alt = len(alternatives)

    words = []
    times = []
    
    for alternative in range(0, num_alt):
        cur_alternative = alternatives[alternative]['alternatives'][0]
        cur_timestamps = cur_alternative['timestamps']

        for cur_word in range(0, len(cur_timestamps)):
            words.append(cur_timestamps[cur_word][0])
            times.append(cur_timestamps[cur_word][1:3])
            
    snip_names = snippet_saver(audio_file, save_dir, prefix, times)
    cap_words = [element.upper() for element in words]
    
    snippet_dict = dict(zip(snip_names, cap_words))
    
    return snippet_dict

            


[docs]def create_phon_dict(cmu_dict):
    """
    Takes the current phonetic dictionary from CMU and creates a dictionary that links
    each word in the dictionary to a list of phonemes for that word.

    Args:
        cmu_dict (str): File location for current CMU phonetic dictionary.

    Returns:
        dict: Dictionary with words as keys and lists of phonemes for each word as values.
    """
    import re
    
    phon_dict = {}
    _rx_alpha = re.compile(r"[a-zA-Z]")

    with open(cmu_dict) as file:
        for line in file:
            if line[0:3] == ";;;":
                continue

            if _rx_alpha.match(line[0]):
                vals = line.split()
                phon_dict[vals[0]] = vals[1:]
                
    return phon_dict




[docs]def phoneme_matcher(audio_dict, cmu_dict):
    """
    Creates a nested dictionary that details which individual words are in specific audio files
    and uses the CMU phonetic dictionary to show which phonemes are in the words that
    correspond to each audio file.

    Args:
        audio_dict (dict): Dictionary with audio file names as keys and corresponding word values.
        cmu_dict (dict): CMU dictionary of words with phonemes.

    Returns:
        dict: Nested dictionary matching audio file names to words and their phonemes.
    """

    words = list(audio_dict.values())
    files = list(audio_dict.keys())
    
    word_phones = [cmu_dict[word] for word in words]
    
    phon_dict = dict(zip(words, word_phones))
    
    word_phon_dict = {}
    
    for cur_word in range(0, len(words)):
        word_phon_dict[files[cur_word]] = {words[cur_word]: phon_dict[words[cur_word]]}
        
    return word_phon_dict

   
    

[docs]def snippet_saver(audio_file, save_dir, prefix, timestamps):
    """
    Takes a given audio file along with the timestamps for when each word occurs
    in the audio and creates individual audio files for each word. Then returns names 
    of audio files that were created.

    Args:
        audio_file (str): Location of audio file.
        save_dir (str): Location in which audio snippets will be saved.
        prefix (str): Prefix to be given to each individual audio snippet file.
        timestamps (list): Timestamps for each individual word found in the audio.

    Returns:
        list: Names of files created.
    """

    from pydub import AudioSegment

    audio = AudioSegment.from_file(audio_file, format = 'wav')
    num_snips = len(timestamps)
    snip_names = []
    
    for cur_word in range(0, num_snips):
        begin = timestamps[cur_word][0] * 1000
        end = timestamps[cur_word][1] * 1000
        cur_snippet = audio[begin:end]
        snip_names.append(save_dir + prefix + '_' + str(cur_word) + '.wav')
        cur_snippet.export(snip_names[cur_word], format = 'wav')
    
    return(snip_names)
