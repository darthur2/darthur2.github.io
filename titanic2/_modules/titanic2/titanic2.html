"""
The titanic2 package includes a suite of functions that use training data sets for both survivors and non-survivors of the Titanic to perform feature engineering, fit a random forest model, and make predictions for a test data set.
"""

import pandas as pd

from sklearn.ensemble import RandomForestClassifier as RFC
from sklearn.ensemble import RandomForestRegressor as RFR


[docs]def Extract_Features(file_name):
    """ Reads in the Titanic csv file and extracts features needed to make predictions.
    
    Args:
        file_name (str): Location of csv file that contains variables for the Titanic analysis.
    

    Returns:
        pandas.DataFrame: Data frame containing extracted features used for fitting the Random Forest needed to make predictions.
    """
    
    data_frame = pd.read_csv(file_name)
    sex_dummies = pd.get_dummies(data_frame['Sex'])
    class_dummies = pd.get_dummies(data_frame['Pclass'], prefix = 'Class')
    sibsp_dummies = pd.get_dummies(data_frame['SibSp'], prefix = 'SibSp')
    parch_dummies = pd.get_dummies(data_frame['Parch'], prefix = 'Parch')
    
    titles = list(map(lambda x: x.split(), data_frame['Name']))
    new_titles = [""]*data_frame.shape[0]
    
    for i in range(0, data_frame.shape[0]):
        if "Mr." in titles[i]:
            new_titles[i] = "Mr"
        elif "Mrs." in titles[i]:
            new_titles[i] = "Mrs"
        elif "Miss." in titles[i]:
            new_titles[i] = "Miss"
        elif "Master." in titles[i]:
            new_titles[i] = "Master"
        else:
            new_titles[i] = "Rare"
            
    title = pd.get_dummies(new_titles)
            
    new_dat = pd.concat([sex_dummies, class_dummies, sibsp_dummies, 
                         parch_dummies, data_frame['Age'],
                        data_frame['Fare'], title], axis = 1)
    
    new_dat['Fare'] = new_dat['Fare'].fillna(new_dat['Fare'].median())
    
    age_dat = new_dat[new_dat['Age'].notnull()]
    noage_dat = new_dat[new_dat['Age'].isnull()]
    
    age_resp = age_dat['Age'].as_matrix()
    age_pred = age_dat.drop(['Age'], axis = 1).as_matrix()
    noage_pred = noage_dat.drop(['Age'], axis = 1).as_matrix()
    
    rf = RFR(n_estimators = 300)
    rf_age = rf.fit(age_pred, age_resp)
    rf_preds = pd.DataFrame(rf_age.predict(noage_pred), columns = ['Age'])
    new_dat.loc[new_dat['Age'].isnull(),'Age'] = rf_preds.values
                
    new_dat = new_dat.drop([col for col in new_dat.columns if 'Parch' in col], axis=1)

    return new_dat

     


[docs]def Fit_Random_Forest(features, train_file):
    """ Takes the response from the training data and the features from the training data and trains a Random Forest model and then returns that model for future predictions.
    
    Args:
        features (pandas.DataFrame): Features used to train Random Forest for future predictions.
        train_file (str): Location of file containing training data.
    
    
    Returns:
        rf_model (sklearn.ensemble.RandomForestClassifier): Trained model to be used for predictions.
    """
    
    train_dat = pd.read_csv(train_file)
    response = train_dat['Survived'].as_matrix()
    features = features.as_matrix()

    rf = RFC(n_estimators = 150)
    rf_model = rf.fit(features, response)
    
    return rf_model

    


[docs]def Predict_File(rf_model, test_file, out_file_name):
    """ Takes trained Random Forest model and uses it to make predictions using the features from test_data. Then it outputs the predictions for each passenger in the test data set along with their passenger ID.
    
    Args:
        rf_model (sklearn.ensemble.RandomForestClassifier): Trained model to be used for predictions.
        test_file (str): Name of file location for test data set.
        out_file_name (str): Name of file location for output file containing predictions.
    
    
    Returns:
        Nothing. Creates a file.
    """
    
    data_frame = pd.read_csv(test_file)
    pass_id = pd.DataFrame(data_frame['PassengerId'])
    test_features = Extract_Features(test_file)
    predictions = pd.DataFrame(rf_model.predict(test_features))
    out_file = pd.concat([pass_id, predictions], axis = 1)
    out_file.columns = ['PassengerId', 'Survived']
    
    out_file.to_csv(out_file_name, index = False)
