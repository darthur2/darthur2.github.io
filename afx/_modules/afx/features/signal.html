[docs]def energy(signal):
    """Computes the energy of discrete input `signal`.
    
    Args:
        signal (numpy.ndarray): Input signal.
        
    Returns:
        float: Energy of input signal `signal`.
        
    Raises:
        OutputError: If energy is too large and an overflow occurs.
        
    Examples:
        The energy of the commonly used normalized sinc function is one. We can 
        investigate this claim with the following:
            
        >>> import numpy as np
        >>> from afx.features.signal import energy as energy
        >>> tt = np.arange(-1000, 1000)
        >>> signal = np.sinc(tt)
        >>> energy(signal)
        1.0
        
        Furthermore, for an unnormalized sinc function, the energy should be 
        very close to :math: \pi.
        
        >>> import numpy as np
        >>> from afx.features.signal import energy as energy
        >>> tt = np.arange(-20000, 20000)
        >>> signal = np.sinc(tt/np.pi)
        >>> np.round(energy(signal), 5)
        3.14154
    """
    
    # Convert input to a numpy array
    
    signal = np.array(signal).astype(np.float)
    
    result = np.sum(np.abs(signal)**2)
    
    # Throws an exception if energy is infinity
    
    if result == np.inf:
        raise OutputError("Computed energy too large. Consider calculating energy "
                         "for smaller segment of signal or standardizing.")
        
    return result


@vd.validate_array({0: ["nolen", "numeric", "empty", "isnan", "isinf"]})
def formant_peak(signal, min_peak_dist=1):
    """Simple peak picking algorithm to see if proposed value for peak is bigger than 
    neighboring values on each side. Code was adapted from a Matlab function that is 
    part of the COVAREP project. For more information, you can go to 
    `here <https://github.com/covarep/covarep/blob/master/formant/formant_CGDZP.m>`_.
    
    Args:
        signal (numpy.ndarray): Signal for which peaks are to be found.
        min_peak_dist (int, float, str): Minimum distance between peaks. Defaults to 1.
        
    Returns:
        numpy.ndarray: Array of indices for locations of peaks in `signal`.
        
    Raises:
        PeakError: If peaks weren't found in `signal`.
        TypeError: If `min_peak_dist` is not a valid integer, float, or string.
        ValueError: If `min_peak_dist` is less than or equal to zero or if input is too short.
        
    Examples:
        We can test this function using a simple sine wave.
        
        >>> import numpy as np
        >>> from afx.features.signal import formant_peak as formant_peak
        >>> tt = np.arange(0, 10000)/100
        >>> sine_wave = np.sin(tt)
        >>> formant_peak(sine_wave)
        array([ 157, 785, 1414, 2042, 2670, 3299, 3927, 4555, 5184, 5812, 6440,
               7069, 7697, 8325, 8954, 9482])
    """
    
    # Checks input for min_peak_dist
    
    try:
        np.float(min_peak_dist)
    except ValueError:
        raise TypeError("Minimum peak distances must be a valid integer or float or a string "
                         "that corresponds to a valid number.")
        
    if min_peak_dist <= 0:
        raise ValueError("Minimum peak distance must be greater than zero.")
    
    # Converts input into a numpy array
    
    signal = np.array(signal).astype(np.float)
    
    # Runs through each possible peak and checks if it meets criteria to be a peak
    
    len_signal= len(signal)
    
    if len_signal <= 11:
        raise ValueError("Input signal should contain at least 12 values in order to "
                         "search for peaks.")
    
    peak_index = []

    for kk in range(5, len_signal-6):
        if (signal[kk] >= signal[kk-1] and signal[kk] >= signal[kk+1])and \
            (signal[kk] >= signal[kk-2] and signal[kk] >= signal[kk+2]) and \
            (signal[kk] > signal[kk-3] and signal[kk] > signal[kk+3]) and \
            (signal[kk] > signal[kk-4] and signal[kk] > signal[kk+4]) and \
            (signal[kk] > signal[kk-5] and signal[kk] > signal[kk+5]):
            peak_index.append(kk)
            
    len_peak_ind = len(peak_index)
    
    if len_peak_ind == 0:
        raise PeakError("No peaks were found. Peaks must be greater than five "
                        "of the closest neighboring values.")
        
    kk = 2
    
    # Goes through peak candidates and combines candidates if they are too close together
    # Then returns a new list of peak indices
    
    while kk <= len_peak_ind:
        if (peak_index[kk-1] - peak_index[kk-2]) < min_peak_dist:
            
            peak_index[kk-1] = np.int(np.round((peak_index[kk-1] * signal[kk-1] + peak_index[kk-2] * 
                      signal[kk-2]) / (signal[kk-1] + signal[kk-2])))
            
            peak_index = peak_index[0:(kk-2)] + peak_index[(kk-1):len_peak_ind]
            kk = kk - 1
            len_peak_ind = len_peak_ind - 1
            
        kk = kk + 1
    
    return np.array(peak_index)


@vd.validate_array({0: ["nolen", "numeric", "empty", "isnan", "isinf"]})
def formants(signal, sample_rate, frame_size=30, frame_shift=10):
    """Computes the first 5 formants of input `signal`. Code was adapted from 
    a Matlab function that is part of the COVAREP project. Further details can 
    be found `here <https://github.com/covarep/covarep/blob/master/formant/formant_CGDZP.m>`_.
    
    Args: 
        signal (numpy.ndarray): Input signal (speech signal).
        sample_rate (int, float): Sample rate of `signal`.
        frame_size (int, float): Size of frames (defaults to 30).
        frame_shift (int, float): Duration between two successive frames (defaults to 10).
        
    Returns:
        numpy.ndarray: Two-dimensional array containing values for first 5 formants of `signal`.
        
    Raises: 
        ValueError: If `frame_size` or `frame_shift` result in window that is too large.
        
    Examples:
        Assuming that `signal` is a one-dimensional :class:`numpy.ndarray` from an audio
        file that was sampled at 16000 Hz, we can use :func:`formants` to track the first 
        five formants with the following:
            
        >>> from afx.features.signal import formants as formants
        >>> sample_rate = 16000
        >>> formants(signal, sample_rate)
    """
    
    # Ensures that input is valid
    
    if sample_rate <= 0:
        raise ValueError("Sample rate should be a positive number.")
    
    # Sets up variables for formant tracking and checks if values are valid
    
    num_formants = 5
    frame_size = np.int(sample_rate/1000*frame_size)
    
    if frame_size > len(signal)-1 or frame_size == 0:
        raise ValueError("Value for `frame_size` resulted in window that is either zero "
                         "or is larger than that input signal.")
        
    frame_shift = np.int(sample_rate/1000*frame_shift)
    
    if (frame_size + frame_shift) > len(signal) or frame_shift == 0:
        raise ValueError("Value for `frame_shift` is such that the window does not shift "
                         "or window is larger than input signal.")
        
    sr_lowres = 2048
    view_range = np.round(sr_lowres/3.2)
    R = 1.12
    Rfix = R
    max_formant_delta = 250    # Discontinuity constraint (in Hz) in formant tracks
    
    n = np.arange(frame_size-1)    # Zero phase data has length of N-1
    
    scaling_freq_axis = np.arange(0.5, 1.5 + 1/(view_range-1), 1/(view_range-1))
    scaling_freq_axis = scaling_freq_axis - 0.25
    
    size_wave = len(signal)
    num_frames = np.int(floor((size_wave - frame_size)/frame_shift))
    
    t_analysis = np.zeros((num_frames,))
    black_win = np.blackman(frame_size)
    
    # Finds the formants (peaks of the chirp group delay) for each window
    
    formant_peaks = np.empty((0, num_formants)).astype(np.int)
    
    for kk in range(num_frames):
        
        speech_data = signal[(kk*frame_shift):(kk*frame_shift+frame_size)]
        window_data = speech_data*black_win
        
        # Obtains zero-phase version of windowed data
        
        zero_phase_data = np.real(sp.ifft(np.abs(sp.fft(np.diff(window_data)))))
        
        num_peaks = 0
        R = Rfix
        
        # Tries to find number of peaks specified by num_formants
        # Changes R value used in exponential_envelope until this happens (if possible)
        
        while num_peaks != num_formants and R > 1.01 and R < 1.25:
            
            # Chirp z-transform calculation using fft
            # Multiplication with exponential function is sufficient
            
            exponential_envelope = np.exp(np.log(1/R)*n)    # Needed for computation of z-transform with fft
            fourier_trans = sp.fft(zero_phase_data*exponential_envelope, sr_lowres)
            angle_FFT = np.angle(fourier_trans[0:int(view_range)])
            chirp_group_delay = -np.diff(angle_FFT)
            peak_index = formant_peak(chirp_group_delay)
            num_peaks = len(peak_index)
            
            if num_peaks > num_formants and R >= Rfix:
                R = R + 0.01
                peak_index =peak_index[0:num_formants]
            elif num_peaks < num_formants and R <= Rfix:
                R = R - 0.01
                peak_index = np.append(peak_index, np.zeros(num_formants - num_peaks,)).astype(np.int)
            else:
                break
        
        # Grabs first num_formants number of peaks or adds in zeroes if it wasn't able to find them
        
        if num_peaks > num_formants:
            peak_index = peak_index[0:num_formants]
        elif num_peaks < num_formants:
            peak_index = np.append(peak_index, np.zeros(num_formants - len(peak_index),)).astype(np.int)
            
        formant_peaks = np.reshape(np.append(formant_peaks, peak_index).astype(np.int), (kk+1, num_formants))
        
        t_analysis[kk] = np.round((kk * frame_shift + kk * frame_shift + frame_size)/2)/sample_rate
    
    # Scales peak indices to the appropriate frequency range for formants
    
    # 1 needs to be added because Python is zero indexing whereas Matlab starts its index with 1
    # Since the formants are related to peak position, we add 1 to account for this
    
    # Also we've added a small decimal to account for differences in rounding for Python and Matlab
    # Still doesn't make answers match up perfectly but they are a lot closer
    # Doing this turns zero values into 8 so we reset them to be 0
    
    formant_peaks = np.round((formant_peaks + 1.000001) * sample_rate / sr_lowres).astype(np.int)
    formant_peaks[formant_peaks == 8] = 0
    
    if True:
        
        # Form a matrix of distance cost
        
        formant_peaks_cost = formant_peaks*0
        
        for kk in range(2, num_frames-2):
            pre_pre_peaks = formant_peaks[kk-2,:]
            post_post_peaks = formant_peaks[kk+2,:]
            pre_peaks = formant_peaks[kk-1,:]
            post_peaks = formant_peaks[kk+1,:]
            current_peaks = formant_peaks[kk,:]
            
            # Cost corresponds to difference from closest match to neighbors
            
            current_peaks_cost = current_peaks*0
            
            for mm in range(num_formants):
                
                if current_peaks[mm] == 0:
                    current_peaks_cost[mm] = sample_rate/2
                else:
                    distance_array_pre = np.sort(np.abs(pre_peaks - current_peaks[mm]))
                    distance_array_post = np.sort(np.abs(post_peaks - current_peaks[mm]))
                    distance_array_prepre = np.sort(np.abs(pre_pre_peaks - current_peaks[mm]))
                    distance_array_postpost = np.sort(np.abs(post_post_peaks - current_peaks[mm]))
                    all_distances = np.array([(distance_array_pre[0] + distance_array_post[0])/2, 
                                              distance_array_pre[0], distance_array_post[0]])
                    all_distances2 = np.array([(distance_array_prepre[0] + distance_array_postpost[0])/2, 
                                              distance_array_prepre[0], distance_array_postpost[0]])
                    current_peaks_cost[mm] = np.min(np.append(all_distances, all_distances2))
            formant_peaks_cost[kk,:] = current_peaks_cost
        
        # Make decision based on costs
        
        for kk in range(num_frames):
            for mm in range(num_formants):
                if formant_peaks_cost[kk,mm] > max_formant_delta:
                    formant_peaks[kk,mm] = 0
    
    # Here we try to replace possible continuation values instead of zero values
            
    if True:
        for kk in range(1, num_frames-1):
            current_peaks = formant_peaks[kk,:]
            index = np.nonzero(current_peaks)    # Finds non-zero elements
            nonzero_formants = current_peaks[index]
            num_nonzero_formants = len(index[0])
            num_zero_formants = num_formants - num_nonzero_formants
            
            if num_nonzero_formants < num_formants and len(index[0]) != 0:
                possible_values = np.sort(np.append(formant_peaks[kk-1,:], formant_peaks[kk+1,:]))
                
                # Discards zero entries
                
                while (len(possible_values) > 0) and (possible_values[0] == 0):
                    possible_values = possible_values[1:]
                
                # To hold candidates for zero valued formants
                
                possible_candidates = np.empty((0,))
                
                # It is somewhat difficult to sythensize data to test some of these conditions
                
                for mm in range(len(possible_values)):
                    
                    distance_array = np.sort(np.abs(nonzero_formants - possible_values[mm]))
                    
                    if len(distance_array) == 0: # pragma: no cover
                        possible_candidates = np.append(possible_candidates, possible_values[mm]).astype(np.int)
                    elif distance_array[0] > max_formant_delta:    # Possible value not found in current vector
                        possible_candidates = np.append(possible_candidates, possible_values[mm]).astype(np.int)
                    
                # This chooses among possible candidates
                
                len_possible_candidates = len(possible_candidates)
                
                if len_possible_candidates <= num_zero_formants: # pragma: no cover
                    current_peaks = np.sort(np.append(np.append(nonzero_formants, possible_candidates), 
                                              np.zeros((num_zero_formants - len_possible_candidates,)))).astype(np.int)
                elif num_zero_formants == 1:    # Most common case
                    sort_index = np.argsort(np.diff(possible_candidates))
                    current_peaks = np.sort(np.append(nonzero_formants, possible_candidates[sort_index[0]]))
                elif num_zero_formants < len_possible_candidates: # pragma: no cover
                    possible_candidates = possible_candidates[0:num_zero_formants]
                    current_peaks = np.sort(np.append(nonzero_formants, possible_candidates))
            
                formant_peaks[kk,:] = current_peaks
            
    return formant_peaks
