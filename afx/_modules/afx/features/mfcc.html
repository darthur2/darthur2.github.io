[docs]def get_filterbanks(nfilt=20, nfft=512, sample_rate=16000, low_freq=0, high_freq=None):
    """Compute a Mel-filterbank. The filters are stored in the rows, the columns correspond
    to fft bins. The filters are returned as an array of size `nfilt` * (`nfft`/2 + 1). Code was adapted from
    `here <https://github.com/jameslyons/python_speech_features>`_.
    
    Args:
        sample_rate (int, float): Sample rate of the signal we are working with.
        nfilt (int): Number of filters in the filterbank, default 26.
        nfft (int): FFT size. Default is 512.
        low_freq (int, float): Lowest band edge of mel filters. In Hz, default is 0.
        high_freq (int, float): Highest band edge of mel filters. In Hz, default is `sample_rate`/2
        
    Returns
        numpy.ndarray: A class:`numpy.ndarray` of size `nfilt` * (`nfft`/2 + 1) containing filterbank. Each row holds 1 filter.
        
    Examples:
        Filterbanks for calculating the Mel-Frequency Cepstral Coefficients can be obtained
        in the following way:
            
        >>> import matplotlib.pyplot as plt
        >>> import afx.features.mfcc as mfcc
        >>> fil_banks = mfcc.get_filterbanks(nfilt=10, nfft=258)
        >>> for fbank in fil_banks:
        ...     plt.plot(fbank)
        >>> plt.title('Filterbanks')
        >>> plt.xlabel('Frequency')
        
    .. figure::  figures/filterbanks.png
       :align:   center
    """

    # Dictionary of paramters for value testing
    
    parms = {'nfilt': nfilt, 'nfft': nfft, 'sample_rate': sample_rate}
    
    # Some parameters must be 1 or greater
    
    for parm in parms:
        if parms[parm] <= 0:
            raise ValueError(parm + " should be at least 1.")
    
    # Choose Nyquist Frequency has default for high_freq
    
    if high_freq is None:
        high_freq = sample_rate/2
    
    # Dictionary of frequency parameters for value testing
    
    freqs = {'low_freq': low_freq, 'high_freq': high_freq}
    
    # Checks values for the frequency parameters
    
    for freq in freqs:
        if freqs[freq] < 0:
            raise ValueError(freq + " should not be negative.")
    
    if high_freq <= low_freq:
        raise ValueError("high_freq should be greater than low_freq.")

    # Compute points evenly spaced in mels
    
    lowmel = hz2mel(low_freq)
    highmel = hz2mel(high_freq)
    melpoints = np.linspace(lowmel,highmel,nfilt+2)
    
    # Our points are in Hz, but we use fft bins, so we have to convert
    # from Hz to fft bin number
    
    fft_bin = np.floor((nfft+1)*mel2hz(melpoints)/sample_rate)
    
    # Makes sure that high_freq isn't set too high
    
    if any(fft_bin[i] > nfft//2+1 for i in range(len(fft_bin))):
        raise IndexError("Value for high_freq results in fft bin number(s) that are larger than "
                         "nfft/2 + 1. Consider value for high_freq that is closer to Nyquist Frequency.")
    
    # Memory error occurs if nfilt and nfft are too big
    
    try:
        fbank = np.zeros([nfilt,nfft//2+1])
    except MemoryError:
        raise MemoryError("Values for nfilt and nfft result in matrix that is too large. Consider smaller values.")
    
    # Creates filter banks
    
    for j in range(0, nfilt):
        for i in range(int(fft_bin[j]), int(fft_bin[j+1])):
            fbank[j,i] = (i - fft_bin[j]) / (fft_bin[j+1]-fft_bin[j])
        for i in range(int(fft_bin[j+1]), int(fft_bin[j+2])):
            fbank[j,i] = (fft_bin[j+2]-i) / (fft_bin[j+2]-fft_bin[j+1])
            
    return fbank



@vd.validate_array({0: ["nolen", "numeric", "empty", "isnan", "isinf"]})
def fbank(signal, sample_rate=16000, win_len=0.025, win_step=0.01,
          nfilt=26, nfft=512, low_freq=0, high_freq=None, pre_emph=0.97,
          win_func=lambda x:np.ones((x,))):
    """Compute Mel-filterbank energy features from an audio signal. Code was adapted from
    `here <https://github.com/jameslyons/python_speech_features>`_.
    
    Args:
        signal (numpy.ndarray): Audio signal from which to compute features. Should be an N*1 :class:`numpy.ndarray`.
        sample_rate (int, float): Sample rate of the signal we are working with.
        win_len (float): Length of the analysis window in seconds. Default is 0.025s (25 milliseconds)
        win_step (float): Step between successive windows in seconds. Default is 0.01s (10 milliseconds)
        nfilt (int, float): Number of filters in the filterbank, default 26.
        nfft (int, float): FFT size. Default is 512.
        low_freq (int, float): Lowest band edge of mel filters. In Hz, default is 0.
        high_freq (int, float): Highest band edge of mel filters. In Hz, default is `sample_rate`/2
        pre_emph (float): Apply preemphasis filter with `pre_emph` as coefficient. 0 is no filter. Default is 0.97. 
        win_func (function): Analysis window to apply to each frame. By default no window is applied.
        
    Returns:
        numpy.ndarray: Numpy array of size (num_frames by `nfilt`) containing features. Each row holds 1 feature vector.
    """
    
    if high_freq is None:
        high_freq = sample_rate/2
        
    signal = preemphasis(signal, pre_emph)
    frames = framesig(signal, win_len*sample_rate, win_step*sample_rate, win_func)
    pspec = powspec(frames, nfft)
        
    fb = get_filterbanks(nfilt, nfft, sample_rate, low_freq, high_freq)
    feat = np.dot(pspec, fb.T) # compute the filterbank energies
    feat = np.where(feat == 0, np.finfo(float).eps, feat) # if feat is zero, we get problems with log
    
    return feat


@vd.validate_array({0: ["nolen", "numeric", "empty", "isnan", "isinf"]})
def framesig(signal, frame_len, frame_step, winfunc=lambda x:np.ones((x,))):
    """Frame `signal` into overlapping frames. Code was adapted from
    `here <https://github.com/jameslyons/python_speech_features>`_.
    
    Args:
        signal (numpy.ndarray): Audio signal from which to compute features. Should be an N*1 :class:`numpy.ndarray`.
        frame_len (int, float): Length of each frame measured in samples.
        frame_step (int, float): Number of samples after the start of the previous frame that the next frame should begin.
        win_func (function): Analysis window to apply to each frame. By default no window is applied.
    
    Returns:
        numpy.ndarray: A class:`numpy.ndarray` of frames. Size is `num_frames` by `frame_len`.
    """
    
    parms = {'frame_len': frame_len, 'frame_step': frame_step}
    
    # Some parameters must be 1 or greater
    
    for parm in parms:
        if parms[parm] <= 0.5:
            raise ValueError(parm + " should be greater than 0.5. Otherwise it will "
                             "round down to zero.")
    
    slen = len(signal)
    frame_len = int(round_half_up(frame_len))
    frame_step = int(round_half_up(frame_step))
    
    if slen <= frame_len:
        numframes = 1
    else:
        numframes = 1 + int(math.ceil((1.0*slen - frame_len)/frame_step))


            
    padlen = int((numframes-1)*frame_step + frame_len)
    
    try:
        zeros = np.zeros((padlen - slen,))
    except MemoryError:
        raise MemoryError("Values for frame_step and frame_len resulted in frame matrix "
                          "that is too large. Consider decreasing their values.")
    
    padsignal = np.concatenate((signal, zeros))

    indices = np.tile(np.arange(0,frame_len), (numframes,1)) + np.tile(np.arange(0, numframes*frame_step ,frame_step), (frame_len, 1)).T
    indices = np.array(indices, dtype=np.int32)
    frames = padsignal[indices]
    
    try:
        winfunc(frame_len)
    except (ValueError, TypeError):
        raise WindowError("Window function can only accept one argument (int): frame_len")
        
    win = np.tile(winfunc(frame_len), (numframes,1))
    
    try:
        win_frames = frames*win
    except (TypeError, ValueError):
        raise WindowError("Window produced with win_func not valid. win_func must "
                          "produce int, float, or an array of either int or float.")
    
    return win_frames



[docs]def hz2mel(hz):
    """Convert a value in Hertz to Mels. For more information, you can go to this
    `tutorial <http://www.practicalcryptography.com/miscellaneous/machine-learning/guide-mel-frequency-cepstral-coefficients-mfccs/>`_.
    
    Args:
        hz (int, float, numpy.ndarray): A value in Hz. This can also be a class:`numpy.ndarray`, conversion proceeds element-wise.
        
    Returns:
        numpy.ndarray: A value in Mels. If an array was passed in, an identical sized array is returned.
    """
    
    # Special validation for this function since it can take int or float
    
    if isinstance(hz, (list, np.ndarray)):
        try:
            np.array(hz).astype(np.float)
        except ValueError:
            raise ValueError("Input array must be a flat, non-empty numeric array.")
        if len(hz) == 0:
            raise ValueError("Input array is empty")
        elif np.any(np.isnan(hz)):
            raise ValueError("Input array contains values that are NaN.")
        elif np.any(np.isinf(hz)):
            raise ValueError("Input array contains values that are Inf.")
        elif np.any([hz[i] == -700 for i in range(len(hz))]):
            raise ValueError("Some values in input are -700 which results in infinite value.")
    else:
        if hz == -700:
            raise ValueError("Value for hz is -700 which results in infinite value.")
                
    hz = np.array(hz).astype(np.float)
    
    return 2595 * np.log10(1+hz/700)



@vd.validate_array({0: ["nolen", "numeric", "empty", "isnan", "isinf"]})
def lifter(cepstra, L=22):
    """Apply a cepstral lifter to the matrix of `cepstra`. This has the effect of increasing the
    magnitude of the high frequency DCT coeffs. Code was adapted from
    `here <https://github.com/jameslyons/python_speech_features>`_.
    
    Args: 
        cepstra (numpy.ndarray): Matrix of mel-cepstra, will be `num_frames` * `num_cep` in size.
        L (int, float): Liftering coefficient to use. Default is 22. L <= 0 disables lifter.
        
    Returns:
        numpy.ndarray: Liftered matrix of mel-cepstra.
    """
    
    if L > 0:
        nframes, ncoeff = np.shape(cepstra)
        n = np.arange(ncoeff)
        lift = 1 + (L/2)*np.sin(np.pi*n/L)
        return lift*cepstra
    else:
        # values of L <= 0, do nothing
        return cepstra


@vd.validate_array({0: ["nolen", "numeric", "empty", "isnan", "isinf"]})
def magspec(frames, nfft):
    """Compute the magnitude spectrum of each frame in frames. If frames is an NxD matrix, output will be Nx(NFFT/2+1).
    
    Args:
        frames (numpy.ndarray): The array of frames. Each row is a frame.
        nfft (int, float): The FFT length to use. If `nfft` > frame_len, the frames are zero-padded.
        
    Returns:
        numpy.ndarray: If `frames` is an NxD matrix, output will be Nx(`nfft`/2+1). 
            Each row will be the magnitude spectrum of the corresponding frame.
    """    
    
    if not isinstance(nfft, int):
        raise TypeError("nfft must be an integer.")
    
    warn_message = ('Frame length {0} is greater than FFT size {1}, '
                    'frame will be truncated. Increase `nfft` to avoid.')
    
    try:
        if np.shape(frames)[1] > nfft: 
            warnings.warn(warn_message.format(np.shape(frames)[1], nfft))
    except (IndexError, ValueError):
        raise FrameError("frames argument should be a two-dimensional array.")
         
    try:
        complex_spec = np.fft.rfft(frames, nfft)
    except MemoryError:
        raise MemoryError("Value for nfft is too large.")
        
    return np.absolute(complex_spec)

    `tutorial <http://www.practicalcryptography.com/miscellaneous/machine-learning/guide-mel-frequency-cepstral-coefficients-mfccs/>`_.
    
    Args:
        mel (numpy.ndarray): A value in Mels. This can also be a class:`numpy.ndrray`, conversion proceeds element-wise.
    
    Returns:
        numpy.ndarray: A value in Hertz. If an array was passed in, an identical sized array is returned.
    """
    
    # Special validation for this function since it can take int or float
    
    if isinstance(mel, (list, np.ndarray)):
        try:
            np.array(mel).astype(np.float)
        except ValueError:
            raise ValueError("Input array must be a flat, non-empty numeric array.")
        if len(mel) == 0:
            raise ValueError("Input array is empty")
        elif np.any(np.isnan(mel)):
            raise ValueError("Input array contains values that are NaN.")
        elif np.any(np.isinf(mel)):
            raise ValueError("Input array contains values that are Inf.")
                
    mel = np.array(mel).astype(np.float)
            
    return 700 * (10**(mel/2595)-1)



@vd.validate_array({0: ["nolen", "numeric", "empty", "isnan", "isinf"]})
def mfcc(signal, sample_rate=16000, win_len=0.025, win_step=0.01, num_cep=13,
         nfilt=26, nfft=512, low_freq=0, high_freq=None, pre_emph=0.97, cep_lifter=22,
         win_func=lambda x:np.ones((x,))):
    """Compute the MFCC features from an audio signal. Code was adapted from
    `here <https://github.com/jameslyons/python_speech_features>`_.
    
    Args:
        signal (numpy.ndarray): Audio signal from which to compute features. Should be an N*1 :class:`numpy.ndarray`.
        sample_rate (int, float): Sample rate of the signal we are working with.
        win_len (float): Length of the analysis window in seconds. Default is 0.025s (25 milliseconds)
        win_step (float): Step between successive windows in seconds. Default is 0.01s (10 milliseconds)
        num_cep (int): Number of cepstrum to return, default 13
        nfilt (int): Number of filters in the filterbank, default 26.
        nfft (int): FFT size. Default is 512.
        low_freq (int, float): Lowest band edge of mel filters. In Hz, default is 0.
        high_freq (int, float): Highest band edge of mel filters. In Hz, default is `sample_rate`/2
        pre_emph (float): Apply preemphasis filter with `pre_emph` as coefficient. 0 is no filter. Default is 0.97. 
        cep_lifter (int, float): Apply a lifter to final cepstral coefficients. 0 is no lifter. Default is 22. 
        win_func (function): Analysis window to apply to each frame. By default no window is applied.
    
    Returns:
        numpy.ndarray: A numpy array of size (`num_frames` by `num_cep`) containing features. Each row holds 1 feature vector.
    """      
     
    if num_cep <= 0:
        raise ValueError("num_cep should be greater than 0.")
        
    feat = fbank(signal, sample_rate, win_len, win_step, nfilt, nfft, low_freq, high_freq, pre_emph, win_func)
    feat = np.log(feat)
    feat = dct(feat, type=2, axis=1, norm='ortho')[:,:num_cep]
    feat = lifter(feat, cep_lifter)
    
    return feat


@vd.validate_array({0: ["nolen", "numeric", "empty", "isnan", "isinf"]})
def powspec(frames, nfft):
    """Compute the power spectrum of each frame in `frames`. If `frames` is an NxD matrix, 
    output will be Nx(NFFT/2+1). Code was adapted from
    `here <https://github.com/jameslyons/python_speech_features>`_.
    
    Args:
        frames (numpy.ndarray): The array of frames. Each row is a frame.
        nfft (int, float): The FFT length to use. If `nfft` > frame_len, the frames are zero-padded. 
        
    Returns:
        numpy.ndarray: If `frames` is an NxD matrix, output will be Nx(`nfft`/2+1). 
            Each row will be the power spectrum of the corresponding frame.
    """

    if not isinstance(nfft, int):
        raise TypeError("nfft must be an integer.")
        
    if nfft == 0:
        raise ZeroDivisionError("nfft should not be 0.")
    
    return 1/nfft * np.square(magspec(frames, nfft))


@vd.validate_array({0: ["nolen", "numeric", "empty", "isnan", "isinf"]})
def preemphasis(signal, coeff=0.95):
    """Perform preemphasis on input `signal`. Code was adapted from
    `here <https://github.com/jameslyons/python_speech_features>`_.
    
    Args:
        signal (numpy.ndarray): Audio signal from which to compute features. Should be an N*1 :class:`numpy.ndarray`.
        coeff (float): The preemphasis coefficient. 0 is no filter, default is 0.95.
    
    Returns:
        numpy.ndarray: The filtered signal.
    """    
    
    if not isinstance(coeff, (int, float)):
        raise TypeError("Argument coeff must be an int or a float")
        
    # Values for coeff are typically between 0 and 1
    
    if coeff < 0 or coeff > 1:
        raise ValueError("Argument coeff should be between 0 and 1 inclusive.")
        
    signal = np.array(signal).astype(np.float)
    
    return np.append(signal[0], signal[1:]-coeff*signal[:-1])



[docs]def round_half_up(number):
    """Performs rounding where 0.5 rounds to 1.
    
    Args:
        number (float): Number to be rounded.
        
    Returns:
        float: Rounded number. 
    """
    
    return int(decimal.Decimal(number).quantize(decimal.Decimal('1'), rounding=decimal.ROUND_HALF_UP))
