def petrosian(X, D=None):
    """Compute Petrosian fractal dimension. Petrosian's algorithm translates
    the series into a binary sequence (a vector of +1 and -1 values).
    :math:`\\frac{\log_{10}(n)}{\log_{10}(n) + \log_{10}(\\frac{n}{n + 0.4 N_\Delta})}`.
    This code was adapted from pyEEG, 0.02 r1: http://pyeeg.sourceforge.net/.

    .. note:: The Petrosian fractal dimension is bounded between 1 and 1.05 and
        therefore cannot give values we would believe are close to the actual fractal
        dimension of a time series. However, the differences within that range might be useful
        for investigating certain phenomenon.

    Args:
        X (numpy.ndarray): Single time series.
        D (numpy.ndarray): First order differential sequence of `X`;
            if not provided, `D` is computed by :func:`numpy.diff`.

    Returns:
        float: Petrosian fractal dimension.

    Raises:
        ValueError: If signal contains less than 2 data points.

    Examples:
        We can first simulate a simple sine wave in the following way:

        >>> import numpy as np
        >>> tt = np.arange(-2000, 2000)/1000
        >>> sine_curve = np.sin(2*np.pi*tt)

        A sine wave should have a fractal dimension of 1. Since this is within the
        appropriate range for the Petrosian fractal dimension, we can test this with:

        >>> from afx.features.fractal import petrosian
        >>> np.round(petrosian(sine_curve), 4)
        1.0001

    """

    # If checks pass then we can make sure the input is a numpy.ndarray

    X = np.array(X).astype(np.float)

    # Validating input for D

    if D is not None:
        try:
            np.array(D).astype(np.float)
        except ValueError:
            raise ValueError("`D` must be a flat, non-empty numeric array.")

        # Checking if there is any data in the input signal

        if len(D) == 0:
            raise ValueError("`D` is empty.")

        # Making sure that the time series doesn't have missing values

        if np.any(np.isnan(D)):
            raise ValueError("`D` contains values that are NaN.")

        # Making sure that infinite values are not in the time series

        if np.any(np.isinf(D)):
            raise ValueError("`D` contains values that are Inf.")

        if len(X) != len(D):
            raise ValueError("`X` and `D` must be same length.")

    # Checking length of signal.

    if len(X) <= 1:
        raise ValueError("`X` must have length greater than 1.")

    # If D has been previously calculated, then it can be passed in here
    #  otherwise, calculate it.

    if D is None:
        D = np.diff(X)   # Difference between one data point and the next

    # The old code is a little easier to follow

    N_delta = 0
    for i in range(1, len(D)):
        if D[i]*D[i-1] < 0:   # Number of sign changes in derivative of the signal
            N_delta += 1

    n = len(X)

    # Original code was missing a paranthesis in the denominator

    petrosian_fd = np.log10(n)/(np.log10(n)+np.log10(n/(n+0.4*N_delta)))

    return petrosian_fd

def katz(X):
    """Calculate the Katz fractal dimension.
    `Example <https://www.seas.upenn.edu/~littlab/Site/Publications_files/Esteller_2001.pdf>`_.

    .. note:: The Katz fractal dimension is calculated in such a way that it will not be lower
        than 1. However, for increasingly complex signals, the Katz fractal dimension will
        continue to grow beyond what one would consider to be a reasonable value for the
        dimension of a time series. In addition, the Katz fractal dimension will almost
        always be close to 1 except for more complex signals.

    Args:
        X (numpy.ndarray): Single time series.

    Returns:
        float: Katz fractal dimension; :math:`\\frac{\log(n)}{\log(n) + \log(\\frac{d}{L})}`,
        where :math:`L` is extent of first point to furthest point, :math:`d` is
        length of time series.

    Raises:
        ValueError: If signal contains less than 3 data points.

    Examples:
        The Weierstrass Function has a theoretical fractal dimension of 1.5. A fairly simple
        curve using the Weierstrass Function can be generated as follows:

        >>> gamma = 5
        >>> M = 27
        >>> H = 0.9
        >>> tt = np.arange(0, 2000)/1333
        >>> weierstrass = np.zeros(len(tt))
        >>>
        >>> for i in range(M):
        ...     weierstrass += gamma**(-i*H) * np.cos(2*np.pi*gamma**i*tt)

        Then we can calculate the Katz fractal dimension with the following:

        >>> from afx.features.fractal import katz
        >>> np.round(katz(weierstrass), 4)
        1.0

        We can generate a more complex curve using the Weierstrass Function with `H` of 0.2.

        >>> H = 0.2
        >>> weierstrass = np.zeros(len(tt))
        >>>
        >>> for i in range(M):
        ...     weierstrass += gamma**(-i*H) * np.cos(2*np.pi*gamma**i*tt)

        Now our estimate of the Katz fractal dimension has increased.

        >>> np.round(katz(weierstrass), 4)
        1.0159

        However in order to really see a large value for the Katz fractal dimension, we can
        try producing a time series composed completely of random draws from a Gaussian
        distribution with mean 0 and standard deviation 20.

        >>> np.random.seed(100)
        >>> gaussian_draws = np.random.normal(0, 20, 1000)
        >>> np.round(katz(gaussian_draws))
        1.853

    """

    # The following is adapted code from matlab function Katz_FD.m by Jesus Monge-Alvarez
    # Code from pyEEG module does not calculate 'd' correctly

    # Making sure that input is numpy.ndarray

    X = np.array(X).astype(np.float)

    # Checking to make sure signal is long enough for calculations

    if len(X) <= 2:
        raise ValueError("`X` must have length greater than 2.")

    L = 0
    N = len(X)
    n = N-1

    # Calculating the length of the time series

    for i in range(0, n):
        aux = np.sqrt(1 + ((X[i] - X[i+1])**2))
        L = L + aux
        del aux

    dist = np.zeros(N)

    # Finding the point such that distance from initial point is largest

    for i in range(1, N):
        dist[i] = np.sqrt(i**2 + ((X[0] - X[i])**2))

    d = np.max(dist)

    katz_fd = np.log10(n) / ((np.log10(n) + np.log10(d/L)))

    # Katz sometimes returns high fractal dimensions for really noise signals
    # While high values are theoretically flawed, they still might be useful

    return katz_fd

def hurst(X):
    """Returns the Hurst exponent of the time series
    vector. `Hurst <https://en.wikipedia.org/wiki/Hurst_exponent>`_ is a measure
    of long-term memory in a time series that is *not* due to periodicity. It's
    often used as a measure of how stationary a time series is.

    Args:
        X (numpy.ndarray): Single time series.

    Returns:
        float: Rate at which the autocorrelation in a time series decreases as the
          lags increase. If the series :math:`x(t)` is a self-similar
          fractal, then :math:`x(bt)` is statistically equivalent to :math:`b^Hx(t)`,
          where `H` is the Hurst exponent.

    Raises:
        ValueError: If signal contains less than 21 data points.
        ZeroDivisionError: If variances for lagged differences are zero.

    Examples:
        For stationary, periodic curves, the fractal dimension of a times series can
        be estimated as 2-H where H is the Hurst exponent. We can test this by
        calculating the Hurst exponent for the Weierstrass Function, which has a theoretical
        fractal dimension of 1.5.

        >>> import numpy as np
        >>> from afx.features.fractal import hurst
        >>> gamma = 5
        >>> M = 27
        >>> H = 0.5
        >>> tt = np.arange(0, 2000)/1333
        >>> weierstrass = np.zeros(len(tt))
        >>>
        >>> for i in range(M):
        ...     weierstrass += gamma**(-i*H) * np.cos(2*np.pi*gamma**i*tt)
        >>>
        >>> 2 - np.round(hurst(weierstrass), 2)
        1.47

    """

    # Making sure input time series is numpy.ndarray

    X = np.array(X).astype(np.float)

    if len(X) <= 20:
        raise ValueError("`X` must have length greater than 100.")


    # Create the range of lag values

    lags = range(2, 20)

    # Calculate the array of the variances of the lagged differences

    tau = [np.sqrt(np.std(np.subtract(X[lag:], X[:-lag])))
           for lag in lags]

    # Making sure that zero values aren't being logged

    if any(tau[x] <= 0 for x in range(len(tau))):
        raise ZeroDivisionError("Array of variances for lagged differences"
                                " contains zero values for which the logarithm"
                                " is undefined.")

    # Use a linear fit to estimate the Hurst Exponent

    poly = np.polyfit(np.log(lags), np.log(tau), 1)

    hurst_exp = poly[0]*2.0

    # Return the Hurst exponent from the polyfit output

    return hurst_exp

def hjorth(X, kmax=5):
    """Compute Hjorth fractal dimension of a time series. This code was taken from `pyEEG
    <http://pyeeg.sourceforge.net/>`_. Hjorth FD is essentially re-sampling (or
    decimating) the time series and checking if we have similar curves. The
    similarity measure is a function of the mean value for each version of the
    series. :math:`k_{\mathrm{max}}` is the scale size (or time offset). So the
    algorithm compares `kmax` versions of the epoch for similarity. The first
    version skips every other datapoint, the second skips every 2 datapoints,
    and the `kmax` skips `kmax` datapoints at a time.

    Args:
        X (numpy.ndarray): Single time series.
        kmax (float): Basically the scale size or time offset; creates `kmax`
          versions of your time series. The `k`-th series is every `k`-th time of
          the original series.

    Returns:
        float: Hjorth fractal dimension.

    Raises:
        ValueError: If `kmax` is too large or signal is too short.
        ZeroDivisionError: If means of chunks are zero.

    Examples:
        Higuchi theorized that the fractal dimension of a time series where each
        value is equal to the previous value plus some gaussian noise would be around 1.5.

        Such a time series can be produced with the following:

        >>> import numpy as np
        >>> seed = 5715005
        >>> np.random.seed(seed)
        >>>
        >>> n = 1000
        >>> gauss_series = np.zeros(n)
        >>> gauss_noise = np.random.normal(size = n)
        >>> gauss_series[0] = np.sum(gauss_noise)
        >>>
        >>> for i in range(1, n):
        ...     gauss_series[i] += gauss_series[i-1] + np.random.normal(size = 1)

        Now we can test to see if the Hjorth fractal dimension is close to 1.5.

        >>> from afx.features.fractal import hjorth
        >>> np.round(hjorth(gauss_series), 2)
        1.49

    """

    # Making sure that input time series is numpy.ndarray

    X = np.array(X).astype(np.float)

    # Checking to make sure kmax and signal length are ok

    if kmax <= 2:
        raise ValueError("`kmax` should be greater than 2.")

    if (2 * kmax - 3) > len(X):
        raise ValueError("Length of `X` must be greater than or equal to "
                         "2 * `kmax` - 3.")

    # Using the mean value in this window to compare similarity to other windows

    # One should note that pyEEG module for the Hjorth Fractal Dimension needs
    # to divide by another 'k' after calculating the mean or the result will almost
    # always be zero. This addition was made by noticing the similarity of this algorithm
    # to algorithm that Higuchi uses to calculate the fractal dimension.

    L = []
    x = []
    N = len(X)
    for k in range(1, kmax):
        Lk = []
        for m in range(k):
            Lmk = 0
            for i in range(1, int(floor((N-m)/k))):
                Lmk += np.abs(X[m+i*k] - X[m+i*k-k])

            Lmk = Lmk*(N - 1)/ floor((N - m) / k) / k
            Lk.append(Lmk)

        L.append(np.mean(Lk)/k)
        x.append([np.log(float(1) / k), 1])

    # Making sure that means of chunks aren't zero so that log can be calculated

    if any(L[i] == 0 for i in range(len(Lk))):
        raise ZeroDivisionError("Means for differences in chunks obtained via "
                                "resampling `X` were zero and could not be logged.")
    else:
        L = np.log(L)

    # Numpy least squares solution

    (p, r1, r2, s) = np.linalg.lstsq(x, L)

    hjorth_fd = p[0]

    return hjorth_fd

def higuchi(X, kmax=7):
    """Calculate the Higuchi fractal dimension. The Higuchi exponent
    is considered to be a measure of the irregularity of a time series.
    Original matlab code by Salai Selvam V. can be seen
    `here <https://www.mathworks.com/matlabcentral/fileexchange/30119-complete-higuchi-fractal-dimension-algorithm/content/hfd.m>`_.

    Args:
        X (numpy.ndarray): Single time series.
        kmax (int): Number of times to resample the series.

    Returns:
        float: Higuchi fractal dimension.

    Raises:
        ValueError: If `kmax` is too large or signal is too short.
        ZeroDivisionError: If means of chunks are zero.

    Examples:
        As explained in the examples for :func:`hjorth`, the fractal dimension of an
        autoregressive time series with gaussian noise is theorized to be 1.5.

        We can first create such a time series with the following:

        >>> import numpy as np
        >>> seed = 5715005
        >>> np.random.seed(seed)
        >>>
        >>> n = 1000
        >>> gauss_series = np.zeros(n)
        >>> gauss_noise = np.random.normal(size = n)
        >>> gauss_series[0] = np.sum(gauss_noise)
        >>>
        >>> for i in range(1, n):
        ...     gauss_series[i] += gauss_series[i-1] + np.random.normal(size = 1)

        Now we can test to see if the Higuchi fractal dimension is close to 1.5.

        >>> from afx.features.fractal import higuchi
        >>> np.round(higuchi(gauss_series), 2)
        1.49

    """

    # Making sure that input time series is numpy.ndarray

    X = np.array(X).astype(np.float)

    # Checking to make sure kmax and signal length are ok

    if kmax <= 1:
        raise ValueError("`kmax` should be greater than 1.")

    if (2 * kmax - 1) > len(X):
        raise ValueError("Length of `X` must be greater than or equal to "
                         "2 * `kmax` - 3.")

    N = len(X)
    Lmk = np.zeros((kmax, kmax))

    for k in range(1, kmax+1):
        for m in range(1, k+1):
            Lmki = 0
            maxI = int(floor((N-m)/k))
            for i in range(1, maxI+1):
                Lmki = Lmki + np.abs(X[m+i*k-1]-X[m+(i-1)*k-1])
            normFactor = (N-1)/(maxI*k)
            Lmk[m-1, k-1] = normFactor * Lmki

    Lk = np.zeros((kmax, 1))

    for k in range(1, kmax+1):
        Lk[k-1, 0] = np.mean(Lmk[range(k), k-1])/k

    # Making sure that means of chunks aren't zero so that log can be calculated

    if any(Lk[i] == 0 for i in range(len(Lk))):
        raise ZeroDivisionError("Means for differences in chunks obtained via "
                                "resampling `X` were zero and could not be logged.")

    lnLk = np.log(Lk)
    lnk = np.log(np.divide(1., range(1, kmax+1)))
    fit = np.polyfit(lnk, lnLk, 1)

    higuchi_fd = float(fit[0])

    # Grab the slope. It is the Higuchi FD

    return higuchi_fd
