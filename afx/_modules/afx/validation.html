[docs]def is_numeric(X):
    """Tests if the specified argument is a valid numeric array (or an object
    that can be cast as one.

    Args:
        X (numpy.ndarray): Numeric array containing signal to be validated.

    Returns:
        bool: True if `X` can be converted to class:`numpy.ndarray`.
    """

    try:
        np.array(X).astype(np.float)
        return True
    except ValueError:
        return False

    

[docs]def no_len(X):
    """Tests if the input has any length.
    
    Args:
        X (numpy.ndarray): Numeric array containing signal to be validated.
        
    Returns:
        bool: True if `X` has length.
    """
    
    try:
        len(X)
        return True
    except TypeError:
        return False


validators = {
    "nolen": no_len,
    "numeric": is_numeric,
    "isnan": lambda a: not np.any(np.isnan(a)),
    "isinf": lambda a: not np.any(np.isinf(a)),
    "empty": lambda a: len(a) > 0
}

messages = {
    "nolen": "Input array is an object without any length.",
    "numeric": "Input array must be a flat, non-empty numeric array.",
    "isnan": "Input array contains values that are NaN.",
    "isinf": "Input array contains values that are Inf.",
    "empty": "Input array is empty."
}


[docs]def validate_array(tests):
    """Validates array arguments passed into `f`.

    Args:
        f: function accepting array arguments.
        tests (dict): keys are class:`int` for positional arguments or class:`str` for
          keyword arguments; values are one of ['numeric', 'isnan', 'isinf', 'empty'].
    """

    def wrap(f):
        def wrapper(*args, **kwargs):
            if tests is not None:
                for k, v in tests.items():
                    if any(vi not in validators for vi in v):
                        raise ValueError("'{}' has unsupported validators.".format(v))

                    if isinstance(k, int) and k < len(args):
                        arg = args[k]
                    elif isinstance(k, str) and k in kwargs:
                        arg = kwargs[k]
                    else:
                        emsg = ("Can't validate argument '{}' not a valid position "
                                "or keyword name.")
                        raise ValueError(emsg.format(k))

                    for vi in v:
                        if not validators[vi](arg):
                            raise ValueError(messages[vi])

            return f(*args, **kwargs)

        return wrapper
    return wrap
