[docs]def word_times(audio_file, auth_file, language = 'en-US_BroadbandModel', watson_learn = False):
    """
    Takes a wav audio file and uses the IBM Watson Speech to Text API to transcribe the audio into text and then extracts words with timestamps.

    Args:
        audio_file (str): Location of wav audio file that is to be transcribed.
        auth_file (str): Location of json file that contains Bluemix credentials needed by IBM to transcribe audio.
        language (str): Language for transcription. For more information see Languages_. Defaults to US English Broadband.
        watson_learn (bool): Indicates whether or not data should be returned to IBM to improve services. Defaults to False.

    Returns:
        dict: Dictionary containing information about the audio transcript along with timestamps for words.

    
    .. _Languages: https://www.ibm.com/watson/developercloud/doc/speech-to-text/input.html#models

    """



[docs]def snippet_saver(audio_file, word_times, target_directory):
    """
    Uses the information provided by the word_times function to segment an audio file into individual words and saves the files.

    Args:
        audio_file (str): Location of wav audio file that is to be segmented.
        word_times (dict): Dictionary from word_times function containing information about words and when they occur in audio_file.
        target_directory (str): Directory in which the resulting audio snippets are to be saved.
        
    Returns:
        Nothing. Saves audio files with words. Names of each file are determined by the words they contain.
    """
