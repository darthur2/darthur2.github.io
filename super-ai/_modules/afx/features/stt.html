[docs]def audio_features(audio_file, auth_file, language = 'en-US_BroadbandModel', watson_learn = False, phonemes = True):
    """Takes an audio file and uses the 
    `IBM Watson Speech to Text API <https://www.ibm.com/watson/developercloud/speech-to-text.html>`_ 
    to transcribe the audio into text and then extracts words along with desired features for each word. 
    Format of audio files must be one of the following:

    - wav
    - flac
    - ogg

    Args:
        audio_file (str): Location of audio file that is to be transcribed.
        auth_file (str): Location of json file that contains Bluemix credentials.
        language (str): Language for transcription. For more information see Languages_.
        watson_learn (bool): Indicates whether or not data should be returned to IBM.
        phonemes (bool): Indicates whether or not phonemes for words should be returned.

    Returns:
        dict: Dictionary containing information about the audio transcript along with desired features for each word.
    
    Raises:
        TypeError: If parameter inputs do not match types specified.
        ValueError: If file extension of audio is not wav, flac, or ogg.
        ValueError: If credentials file is not formatted as json.
        KeyError: If `username` or `password` were not found in json file.
        ValueError: If values for `username` or `password` were not found in json file.
    
    .. _Languages: https://www.ibm.com/watson/developercloud/doc/speech-to-text/input.html#models

    Examples:
        Assuming that `audio.wav` is a wav file and that `auth.json` is a json file with the
        following format:
        
        .. code-block:: json

            {
                "username": "0c61ebc2-fake-json-file-371cea2fa3ff",
                "password": "fakepassword",
            }

        where `password and `username` are your BlueMix Credentials for the Watson API, then
        the audio can be transcribed with the following:

        >>> from afx.features.stt import audio_features as af
        >>> audio_file = 'audio.wav'
        >>> auth_file = 'auth.json'
        >>> af(audio_file, auth_file)  

    """
    
    # Checking to make sure that inputs are correct types
    
    if not isinstance(audio_file, str):
        raise TypeError("Expecting object of type `str`")
    
    if not isinstance(auth_file, str):
        raise TypeError("Expecting obect of type `str`")
    
    if not isinstance(language, str):
        raise TypeError("Expecting object of type `str`")
    
    if not isinstance(watson_learn, bool):
        raise TypeError("Expecting object of type `bool`")
    
    if not isinstance(phonemes, bool):
        raise TypeError("Expecting object of type `bool`")
    
    
    # Checking to make sure that audio file is supported
    
    audio_ext = path.splitext(audio_file)[1]
    
    if audio_ext not in ['.wav', '.flac', '.ogg']:
        raise ValueError("Format of provided audio file is not supported")
    
    # Ensures that credentials file is a json file
    
    auth_ext = path.splitext(auth_file)[1]
    
    if auth_ext != '.json':
        raise ValueError("Credentials file must be formatted as json file")
    
    # Opens the json file containing credentials needed by IBM to transcribe
    
    with open(auth_file) as cred_file:
        auth_info = json.load(cred_file)
        
    # Checks to see if password and username exist in json file
        
    if 'username' not in list(auth_info.keys()):
        raise KeyError("Could not find `username` in credentials file")
    
    if 'password' not in list(auth_info.keys()):
        raise KeyError("Could not find `password` in credentials file")
    
    if auth_info['username'] == '':
        raise ValueError("`username` is empty")
    
    if auth_info['password'] == '':
        raise ValueError("`password` is empty")
    
    # Sets username and password

    speech_user = auth_info['username']
    speech_pass = auth_info['password']

    # Creates a speech to text instance
    
    speech_to_text = SpeechToTextV1(
        username = speech_user,
        password = speech_pass,
        x_watson_learning_opt_out = watson_learn
    )
    
    # Grabs the specified language model
    
    speech_to_text.get_model(language)

    # Recognize function reads the audio and returns the transcription

    with open(audio_file, 'rb') as audio:
        speech_results = speech_to_text.recognize(
            audio, content_type = 'audio/' + audio_ext[1:], timestamps = True
        )

    # Grabs all of the results returned by the Speech API
    
    results = speech_results['results']
    num_res = len(results)
    
    # Grabs the words and timestamps from the first result (phrase)
    
    first_result = results[0]['alternatives'][0]
    words = first_result['transcript'].split()
    first_times = first_result['timestamps']
    timestamps = [first_times[cur_word][1:3] for cur_word in range(0,len(first_times))]

    # Gets words and timestamps for additional chunks of phrases that were returned
    
    for result in range(1, num_res):
        cur_result = results[result]['alternatives'][0]
        words += cur_result['transcript'].split()
        cur_times = cur_result['timestamps']
        timestamps += [cur_times[cur_word][1:3] for cur_word in range(0,len(cur_times))]
    
    # Need to convert words to upper case to use the CMU Dictionary
    
    words = [element.upper() for element in words]
    
    # Reads in the specified audio as pydub class AudioSegment
    
    audio = AudioSegment.from_file(audio_file)    
    num_words = len(words)
    audio_snippets = []
    
    # Creates a list of numpy arrays that correspond to words in audio
    
    for cur_word in range(0, num_words):
        begin = timestamps[cur_word][0] * 1000
        end = timestamps[cur_word][1] * 1000
        audio_snippets.append(np.array((audio[begin:end]).get_array_of_samples()))
    
    # Matches words from audio to words in CMU dictionary (if they exist) and gets phonemes
    # If words occur more then once, their information is appended to that word
    
    if phonemes == True:
        word_in_dict = [words[word] in cmu_dict for word in range(0, len(words))]
        words_used = list(itertools.compress(words, word_in_dict))
        stamps_used = list(itertools.compress(timestamps, word_in_dict))
        snippets_used = list(itertools.compress(audio_snippets, word_in_dict))
        word_phones = [cmu_dict[word] for word in words_used]
        
        audio_dict = {}
        
        for word in range(len(words_used)):
            if words_used[word] in list(audio_dict.keys()):
                audio_dict[words_used[word]][0].append(stamps_used[word])
                audio_dict[words_used[word]][1].append(snippets_used[word])
            else:
                audio_dict[words_used[word]] = ([stamps_used[word]], [snippets_used[word]], word_phones[word])
            
    # Matches words from audio to their timestamps and waveforms
    # The code is the exact same as above except for the lack of phonemes
    
    else:
        audio_dict = {}
        
        for word in range(num_words):
            if words[word] in list(audio_dict.keys()):
                audio_dict[words[word]][0].append(timestamps[word])
                audio_dict[words[word]][1].append(audio_snippets[word])
            else:
                audio_dict[words[word]] = ([timestamps[word]], [audio_snippets[word]])
         
    # Returns a dictionary with the specified features
    
    return audio_dict



[docs]def phone_selector(audio_dict, phones):
    """Takes the dictionary of features created by :func:`audio_features` and 
    finds those words that contain the phonemes found in `phones` and returns 
    waveforms for those words.

    Args:
        audio_dict (dict): Dictionary of features for the audio used in :func:`audio_features`.
        phones (list): List of phonemes.

    Returns:
        dict: Dictionary with words that contained phonemes found in `phones` 
        as keys and waveforms as values.

    Raises:
        TypeError: If types of input do not match those specified.
        ValueError: If `audio_dict` is empty.
        ValueError: If none of the phonemes specified were found.
        IndexError: If `audio_dict` is missing phonemes for words.

    Examples:
        Assume that `audio_dict` holds the output from using :func:`audio_features`. 
        Then we can search the dictionary for specific phonemes in the following way:

        >>> from afx.features.stt import phone_selector as phs
        >>> phonemes = ['D', 'IY1', 'AA1', 'W']
        >>> phs(audio_dict, phonemes)
    
    """
    
    # Checks to make sure input are of correct types
    
    if not isinstance(audio_dict, dict):
        raise TypeError("`audio_dict` must be of type `dict`")
    
    if not isinstance(phones, list):
        raise TypeError("`phones` must be a `list` of phonemes")
    
    # Ensures that dictionary is not empty
    
    if len(audio_dict) == 0:
        raise ValueError("`audio_dict` is empty")
        
    # Makes sure that phonemes were supplied
    
    if len(audio_dict[list(audio_dict.keys())[0]]) != 3:
        raise IndexError("Index out of range. `audio_dict` doesn't contain phonemes.")
        
    # Get word keys and phoneme list values from audio_dict
    
    values = list(audio_dict.values())
    keys = list(audio_dict.keys())
    
    # Creates lists of phonemes, both overall and for individual words
    
    word_phones = [values[word][2] for word in range(len(values))]
    all_phones = list(itertools.chain.from_iterable(word_phones))
    
    # Ensures that at least one of the phonemes given is found in at least one word
    
    if any([phones[phone] in all_phones for phone in range(len(phones))]) == False:
        raise ValueError("None of phonemes in phone found in `audio_dict`")
    
    # Get waveform representations for each word
    
    waveforms = [values[wave][1] for wave in range(len(values))]
    
    # Find out which words contain the phonemes that we are looking for
    
    phone_ind = [[[any(phones[phone] in word_phones[word] for phone in range(len(phones)))] for 
      word in range(len(word_phones))][ind][0] for ind in range(len(word_phones))]
    
    # Creates keys of words and values of waveform representations for each word
    
    phone_keys = list(itertools.compress(keys, phone_ind))
    wave_values = list(itertools.compress(waveforms, phone_ind))
    
    phone_dict = dict(zip(phone_keys, wave_values))
    
    return phone_dict




[docs]def pos_selector(audio_dict, parts_of_speech):
    """Takes the dictionary of features created by :func:`audio_features` and 
    finds those words that contain the parts of speech found in `parts_of_speech` 
    and returns the waveforms for those words.

    Args:
        audio_dict (dict): Dictionary of features for the audio used in :func:`audio_features`
        parts_of_speech (list): List of parts of speech.

    Returns:
        dict: Dictionary with words that contained parts of speech found in `parts_of_speech` 
        as keys and waveforms as values.

    Raises:
        TypeError: If types of input do not match those specified.
        ValueError: If `audio_dict` is empty.
        ValueError: If none of the parts of speech listed were found.
        IndexError: If `audio_dict` is missing phonemes for words.
    
    Examples:
        If `audio_dict` is the output from using :func:`audio_features`, then we can search
        the dictionary for words with specific parts of speech in the following way:

        >>> from afx.features.stt import pos_selector as pos
        >>> parts_of_speech = ['liquid', 'stop']
        >>> pos(audio_dict, parts_of_speech)

    """
    
    # Ensures that input is of correct type
    
    if not isinstance(audio_dict, dict):
        raise TypeError("`audio_dict` must of type `dict`")
    
    if not isinstance(parts_of_speech, list):
        raise TypeError("`parts_of_speech` must be a `list` of parts of speech")
    
    # Ensures that dictionary is not empty
    
    if len(audio_dict) == 0:
        raise ValueError("`audio_dict` is empty")
  
    # Makes sure that phonemes were supplied
      
    if len(audio_dict[list(audio_dict.keys())[0]]) != 3:
        raise IndexError("Index out of range. `audio_dict` doesn't contain phonemes.")
 
    # Gets parts of speech values and phoneme keys
    
    pos_values = list(pos_dict.values())
    pos_keys = list(pos_dict.keys())
    
    # Indicates which parts of speech values in pos_values we want to look for
    
    pos_ind = [[[any(parts_of_speech[part_of_speech] in pos_values[key] for part_of_speech in 
                     range(len(parts_of_speech)))] for key in range(len(pos_values))][ind][0] 
        for ind in range(len(pos_values))]
        
    # Creates a list of phonemes that correpond to the parts of speech we are looking for
    
    phones = list(itertools.compress(pos_keys, pos_ind))
    
    # Tries to use phone_selector function. If phone_selector doesn't work, that means that
    # none of the words in audio_dict contain the parts of speech we were looking for
    
    try:
        phone_selector(audio_dict, phones)
    except ValueError:
        raise ValueError("None of parts of speech in `parts_of_speech` found in `audio_dict`")
        
    # Uses phone_selector function to get words and waveforms that we want
    
    parts_of_speech_dict = phone_selector(audio_dict, phones)
    
    return parts_of_speech_dict
